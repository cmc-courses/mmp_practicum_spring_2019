def generate_noise(n_samples, sample_size, min_errors, max_errors):
    max_errors = min(sample_size, max_errors)
    min_errors = min(max(0, min_errors), sample_size)
    noise = u = np.zeros([n_samples, sample_size], dtype=np.int)
    n_errors = np.sort(np.random.randint(min_errors, max_errors + 1, [n_samples]))
    for idx in range(n_samples):
        permutation = np.random.permutation(sample_size)
        noise[idx, permutation[:n_errors[idx]]] = 1
    return noise

def count_statistics(n_samples, b):
    correct, mistake, reject = np.zeros([b.n + 1], dtype=np.int), np.zeros([b.n + 1], dtype=np.int), np.zeros([b.n + 1], dtype=np.int)

    for n_errors in range(0, b.n + 1):
        u = np.random.randint(0, 2, [n_samples, b.k])
        v = b.encode(u)
        noise = generate_noise(n_samples, b.n, n_errors, n_errors)
        w_noised = v ^ noise
        w_recovered = b.decode(w_noised)
        u_recovered = w_recovered[:, :b.k]

        for idx, (noise_size, n_mistakes) in enumerate(zip(np.sum(noise, axis=1), np.sum(u_recovered ^ u, axis=1))):
            if u_recovered[idx][0] < 0:
                reject[noise_size] += 1
            elif n_mistakes == 0:
                correct[noise_size] += 1
            elif n_mistakes > 0:
                mistake[noise_size] += 1
    return correct, mistake, reject

    bar_2 = ax.bar(np.arange(0, mistake.shape[0]), mistake, bottom=correct, label='decoded with mistake')    
    bar_3 = ax.bar(np.arange(0, reject.shape[0]), reject, bottom=correct + mistake, label='decoding rejection')
    ax.legend()
    ax.set_xlabel('number of changed bits', fontsize = 20)
    ax.set_ylabel('number of messages', fontsize = 20)
    
    fig.show()

def measure_time(n_samples, b):
    n_samples, n_errors = n_samples, b.t

    u = np.random.randint(0, 2, [n_samples, b.k])
    v = b.encode(u)
    noise = generate_noise(n_samples, b.n, 0, n_errors)
    w_noised = v ^ noise

    start_time_pgz = time.time()
    w_recovered_pgz = b.decode(w_noised, method='pgz')
    end_time_pgz = time.time()

    start_time_euclid = time.time()
    w_recovered_euclid = b.decode(w_noised, method='euclid')
    end_time_euclid = time.time()

    assert np.sum(w_recovered_pgz != w_recovered_euclid) == 0
    
    u_recovered = w_recovered_pgz[:, :b.k]

    time_pgz, time_euclid = end_time_pgz - start_time_pgz, end_time_euclid - start_time_euclid

    return time_pgz, time_euclid
